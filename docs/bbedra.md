### *Bartłomiej Bedra*

## Przykładowe dane:

```json
{
    "id" : 6523,
    "type" : "heliport",
    "name" : "Total Rf Heliport",
    "latitude_deg" : 40.07080078125,
    "longitude_deg" : -74.9336013793945,
    "elevation_ft" : 11,
    "continent" : "NA",
    "iso_country" : "US",
    "iso_region" : "US-PA",
    "municipality" : "Bensalem",
    "gps_code" : "00A"
}
```
## Map reduce
### Funkcja map

```javascript
var m = function() {
    emit(
        {kraj: this.iso_country}, {count: 1}
    );
};
```

### Funkcja reduce

```javascript
var r = function(key, vals) {
	var count = 0;
    vals.forEach(function (v) {
        count += v['count'];
    });
    return {count: count};
};
```

### Przykładowe wywołanie

```javascript
db.lotniska.mapReduce(m, r, {out: 'lotniska_results'});
```
